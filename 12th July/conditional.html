IF:
        Syntax:
        if(condition){
            true block;
        }

        IF-ELSE:
        if(condition){
            true block;
        }
        else{
            false block;
        }

        IF-ELSEIF 
        if(condition1){
            true block;
        }
        else if(condition2){
            true block;
        }
        else{
            false block;
        }

        NESTED IF

        if(condition){
            if(condition){

            }
        }

        SWITCH
        switch(condition){
            case 1:
            break;

            default:
        }
